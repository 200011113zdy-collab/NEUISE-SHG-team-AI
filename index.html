import os import time import base64 import shutil import subprocess from flask import Flask, request, jsonify from flask_cors import CORS from dotenv import load_dotenv import sys # 加载 .env 文件中的配置 load_dotenv() # 🔑 尝试读取密码 MY_SECRET_KEY = os.getenv("MY_SECRET_KEY") if not MY_SECRET_KEY: print("🚨 致命错误：未找到访问密码或密码为空！") print("👉 请确保在 server.py 同级目录下创建了 .env 文件，并正确填写了 MY_SECRET_KEY=你的密码") sys.exit(1) CLAUDE_PROXY_URL = os.getenv("CLAUDE_PROXY_URL", "") app = Flask(__name__) CORS(app) @app.route('/api/config', methods=['GET']) def get_config(): auth_header = request.headers.get('Authorization', '') if auth_header != f"Bearer {MY_SECRET_KEY}": return jsonify({"error": "访问拒绝：密码错误！"}), 401 return jsonify({"claude_url": CLAUDE_PROXY_URL}) @app.route('/api/claude', methods=['POST']) def call_claude(): auth_header = request.headers.get('Authorization', '') if auth_header != f"Bearer {MY_SECRET_KEY}": return jsonify({"error": "访问拒绝：密码错误！请在设置中填写正确的本地服务密码。"}), 401 data = request.json user_prompt = data.get('prompt', '') files_data = data.get('files', []) search_enabled = data.get('search_enabled', False) if not user_prompt: return jsonify({"error": "请求内容不能为空"}), 400 try: with open("chat_history.txt", "w", encoding="utf-8") as f: f.write(user_prompt) # 📁 处理上传的文件 saved_filenames = [] for idx, file_info in enumerate(files_data): orig_name = file_info.get('name', f'upload_{idx}') ext = os.path.splitext(orig_name)[1] if '.' in orig_name else '.bin' safe_filename = f"temp_upload_{int(time.time())}_{idx}{ext}" file_b64 = file_info.get('data', '') if file_b64: with open(safe_filename, "wb") as f_out: f_out.write(base64.b64decode(file_b64)) saved_filenames.append(safe_filename) my_env = os.environ.copy() my_env['HTTP_PROXY'] = 'http://127.0.0.1:10808' my_env['HTTPS_PROXY'] = 'http://127.0.0.1:10808' file_instruction = ( f"用户上传了以下文件，已保存在当前目录：{', '.join(saved_filenames)}。请使用文件读取工具读取并分析它们。" if saved_filenames else "" ) search_instruction = ( "如有需要，可以使用 web_search 工具搜索最新信息。" if search_enabled else "本次任务无需联网搜索。" ) # 📂 准备专用输出目录：每次请求前清空，确保只收集本次生成的文件 output_dir = os.path.join(os.getcwd(), "claude_output") history_dir = os.path.join(os.getcwd(), "claude_history") for d in [output_dir, history_dir]: os.makedirs(d, exist_ok=True) # 清空上次残留（正常情况下应已被归档，但作为保险） for f in os.listdir(output_dir): os.remove(os.path.join(output_dir, f)) agent_command = ( f"请读取当前目录下的 chat_history.txt，其中包含用户的对话历史。" f"{file_instruction} " f"{search_instruction} " f"根据对话历史，回答用户最后提出的问题。" f"【文件生成规则】如果用户要求生成文件，请遵守以下规则：" f"1. 所有生成的文件必须保存到 ./claude_output/ 目录，文件名要有意义，不要保存到其他位置。" f"2. 生成 .docx 时：使用 python-docx 库，设置中文字体（如宋体/黑体），" f" 合理使用标题样式（Heading 1/2/3）、正文段落、表格，段落间距适当，内容详尽专业。" f"3. 生成 .pptx 时：使用 python-pptx 库，每页有清晰标题，内容条理分明，" f" 设置合适的字体大小和颜色，幻灯片数量不少于8页。" f"4. 生成 .xlsx 时：使用 openpyxl 库，设置列宽、表头加粗，数据格式规范。" f"5. 生成代码文件时：代码完整可运行，有必要的注释。" ).strip() result = subprocess.run( ['claude.cmd', '--dangerously-skip-permissions', '-p'], input=agent_command, capture_output=True, text=True, timeout=600, # 10分钟，支持复杂文档生成 encoding='utf-8', errors='ignore', env=my_env ) # 🧹 清理上传的临时文件 for f_name in saved_filenames: if os.path.exists(f_name): os.remove(f_name) # 📦 读取 claude_output/ 里的文件，文本用 标签，二进制 base64 回传 file_tags = "" binary_files = [] timestamp = time.strftime("%Y%m%d_%H%M%S") BINARY_EXTS = {'.docx', '.xlsx', '.pptx', '.pdf', '.zip', '.png', '.jpg', '.jpeg', '.gif'} for fname in sorted(os.listdir(output_dir)): fpath = os.path.join(output_dir, fname) if not os.path.isfile(fpath): continue ext = os.path.splitext(fname)[1].lower() if ext in BINARY_EXTS: # 二进制文件：base64 编码后单独回传 try: with open(fpath, 'rb') as fh: b64 = base64.b64encode(fh.read()).decode('utf-8') binary_files.append({'name': fname, 'data': b64}) except Exception: pass else: # 文本文件：拼成 标签嵌入回复 try: with open(fpath, 'r', encoding='utf-8', errors='replace') as fh: fc = fh.read() file_tags += f'\n\n{fc}\n\n' except Exception: pass # 📁 归档到 claude_history/，文件名加时间戳前缀防重名 name_base, name_ext = os.path.splitext(fname) archive_path = os.path.join(history_dir, f"{timestamp}_{name_base}{name_ext}") counter = 1 while os.path.exists(archive_path): archive_path = os.path.join(history_dir, f"{timestamp}_{name_base}_{counter}{name_ext}") counter += 1 shutil.move(fpath, archive_path) output = (result.stdout or result.stderr or "").strip() if not output: output = f"[调试] 无输出。退出码={result.returncode}\nSTDOUT={result.stdout!r}\nSTDERR={result.stderr!r}" return jsonify({"response": output.strip() + file_tags, "binary_files": binary_files}) except Exception as e: return jsonify({"error": f"系统错误: {str(e)}"}), 500 if __name__ == '__main__': app.run(host='127.0.0.1', port=5000)
